Introduction to SQL - AGGREGATED DATA USING THE GROUP FUNCTIONS

1). ans--SELECT ROUND(MAX(SAL)) AS "Maximum",
       ROUND(MIN(SAL)) AS "Minimum",
       ROUND(SUM(SAL)) AS "Sum",
       ROUND(AVG(SAL)) AS "Average"
FROM EMP;

2). ans-- SELECT JOB,
       ROUND(MIN(SAL)) AS "Minimum",
       ROUND(MAX(SAL)) AS "Maximum",
       ROUND(SUM(SAL)) AS "Sum",
       ROUND(AVG(SAL)) AS "Average"
FROM EMP
GROUP BY JOB
ORDER BY JOB;

3). ans -- SELECT JOB,
       ROUND(MIN(SAL)) AS "Minimum",
       ROUND(MAX(SAL)) AS "Maximum",
       ROUND(SUM(SAL)) AS "Sum",
       ROUND(AVG(SAL)) AS "Average"
FROM EMP
GROUP BY JOB
ORDER BY JOB;

4). ans--SELECT JOB,
       COUNT(*) AS "Number of Employees"
FROM EMP
GROUP BY JOB
ORDER BY JOB;

5). ans -- SELECT COUNT(DISTINCT MGR) AS "Number of Managers"
FROM EMP;

6). ans -- SELECT MAX(SAL) - MIN(SAL) AS "DIFFERENCE"
FROM EMP;

7). ans-- SELECT MGR,
       MIN(SAL) AS "Lowest Salary"
FROM EMP
WHERE MGR IS NOT NULL
GROUP BY MGR
HAVING MIN(SAL) > 2000
ORDER BY MIN(SAL) DESC;
