Introduction to SQL - JOINS

1). ans-- SELECT EMPNO, ENAME, SAL, DNAME, LOC
FROM EMP
NATURAL JOIN DEPT;

2). ans-- SELECT JOB, MGR, SAL, COMM, DNAME
FROM EMP
NATURAL JOIN DEPT
WHERE JOB = 'SALESMAN';

3). ans-- SELECT ENAME, JOB, DEPTNO, DNAME
FROM EMP
NATURAL JOIN DEPT
WHERE LOC = 'DALLAS';

4). ans-- SELECT E.ENAME AS "Employee",
       E.EMPNO AS "Emp#",
       M.ENAME AS "Manager",
       M.EMPNO AS "Mgr#"
FROM EMP E
JOIN EMP M
ON E.MGR = M.EMPNO;

5). ans-- SELECT E.ENAME AS "Employee",
       E.EMPNO AS "Emp#",
       M.ENAME AS "Manager",
       M.EMPNO AS "Mgr#"
FROM EMP E
LEFT OUTER JOIN EMP M
ON E.MGR = M.EMPNO
ORDER BY E.EMPNO;

6). ans-- DESC SALGRADE;

SELECT E.ENAME,
       E.JOB,
       D.DNAME,
       E.SAL,
       S.GRADE
FROM EMP E
JOIN DEPT D
ON E.DEPTNO = D.DEPTNO
JOIN SALGRADE S
ON E.SAL BETWEEN S.LOSAL AND S.HISAL
ORDER BY E.EMPNO;

7). ans-- SELECT ENAME, DNAME
FROM EMP
RIGHT OUTER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO;

8). ans--SELECT E.ENAME,
