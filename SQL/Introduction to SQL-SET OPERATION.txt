Introduction to SQL - USING THE SET OPERATORS

1). ans -- SELECT JOB,
       SUM(DECODE(DEPTNO,10,SAL)) AS "DEPT 10",
       SUM(DECODE(DEPTNO,20,SAL)) AS "DEPT 20",
       SUM(DECODE(DEPTNO,30,SAL)) AS "DEPT 30",
       SUM(SAL) AS "TOTAL"
FROM EMP
GROUP BY JOB
ORDER BY JOB;

2). ans-- SELECT TO_CHAR(DEPTNO) AS CATEGORY,
       SUM(SAL) AS TOTAL_SALARY
FROM EMP
GROUP BY DEPTNO

UNION ALL

SELECT JOB,
       SUM(SAL)
FROM EMP
GROUP BY JOB

UNION ALL

SELECT 'TOTAL',
       SUM(SAL)
FROM EMP;

3). ans-- SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 20

UNION ALL

SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 10

UNION ALL

SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 30;
